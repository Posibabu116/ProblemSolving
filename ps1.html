Series Calculations

Non-Fibonacci Series

        // Input: Enter no of non-fib series: 10

        n = +prompt("enter no:of series: ");
        let c1;
        a = 0;
        b = 1;
        res = "";
        while (a <= n) {
            for (i = a + 1; i < b; i++) {
                c1++;
                res = res + i + " ";
                if (i == n) {
                    break;
                }
            }
            let c = a + b;
            a = b;
            b = c;
        }
        console.log(res);

        // Output: 4 6 7 9 10

Prime Numbers in Fibonacci Series

        // Input: Enter no of fib-series: 7

        n = +prompt("enter no:of series: ");
        a = 0;
        b = 1;
        i = 1;
        res = "";
        count = 0;
        while (i <= n) {
            if (a > 1) {
                for (let j = 2; j < a; j++) {
                    count = 0;
                    if (a % j == 0) {
                        count++;
                        break;
                    }
                }
                if (count == 0) {
                    res = res + a + " ";
                }
            }
            let c = a + b;
            a = b;
            b = c;
            i++;
        }
        console.log("prime numbers in the fib series are: ", res);

        // Output: prime numbers in the fib series are: 2 3 5

Non-Prime Numbers in Fibonacci Series

        // Input: Enter no of fib-series: 7

        n = +prompt("enter no:of series: ");
        a = 0;
        b = 1;
        i = 1;
        res = "";
        while (i <= n) {
            if (a < 2) {
                res = res + a + " ";
            } else {
                for (let j = 2; j < a; j++) {
                    if (a % j == 0) {
                        res = res + a + " ";
                        break;
                    }
                }
            }
            let c = a + b;
            a = b;
            b = c;
            i++;
        }
        console.log("Non prime numbers in the fib series are: ", res);

        // Output: Non prime numbers in the fib series are: 0 1 1 8
