1. Sum of Odd Digits

            let x = "1633";
            let oddsum = 0;
            for (let i = 0; i < x.length; i++) {
                if (x[i] % 2 == 1) {
                    oddsum += Number(x[i]);
                }
            }
            console.log("Odd sum is:", oddsum);

2. Difference between Even Sum and Odd Sum

            let x = "163382";
            let oddsum = 0;
            let evensum = 0;
            for (let i = 0; i < x.length; i++) {
                if (x[i] % 2 == 1) {
                    oddsum += Number(x[i]);
                } else {
                    evensum += Number(x[i]);
                }
            }
            let d;
            if (oddsum > evensum) {
                d = oddsum - evensum;
                console.log("Difference is ", d);
            } else {
                d = evensum - oddsum;
                console.log("Difference is ", d);
            }
